123 uint32_t convert32(uint32_t x) {
124  static const uint32_t E0 = 0x11111111;
125  static const uint32_t E1 = 0x22222222;
126  static const uint32_t E2 = 0x44444444;
127  static const uint32_t E3 = 0x88888888;
128  int i;
129  uint32_t u, y, z, r = 0;
130  uint64_t v;
131  for (i = 0; i < 8; ++i) {
132  v = x;
133  v *= 9;
134  u = (v >> 32);
135  x = v;
136  r = u ^ (r << 4);
137  }
138  y = r + E0;
139  z = r + E1;
140  r += ((y & z & E2) ^ (z & E3)) >> 2;
141 
142  return r;
143 }

146 uint64_t convert64(uint64_t x) {
147  static const uint64_t F2 = 0x0404040404040404;
148  static const uint64_t F3 = 0x0808080808080808;
149  static const uint64_t F4 = 0x1010101010101010;
150  static const uint64_t F5 = 0x2020202020202020;
151  static const uint64_t F6 = 0x4040404040404040;
152  static const uint64_t F7 = 0x8080808080808080;
153  static const uint64_t Y2 = 0x0101010101010101;
154  static const uint64_t Y3 = 0x0707070707070707;
155  static const uint64_t Y4 = 0x2525252525252525;
156  int i;
157  uint64_t u, y, z, r = 0;
158  unsigned __int128 v;
159  for (i = 0; i < 8; ++i) {
160  v = x;
161  v *= 81;
162  u = (v >> 64);
163  x = v;
164  r = u ^ (r << 8);
165  }
166  y = r + Y4;
167  z = r + (2 * Y4);
168  r += (((y & z & F6) ^ (z & F7)) >> 6) * 37;
169  y = r + Y3;
170  z = r + (2 * Y3);
171  r += (((y & z & F4) ^ (z & F5)) >> 4) * 7;
172  y = r + Y2;
173  z = r + (2 * Y2);
174  r += ((y & z & F2) ^ (z & F3)) >> 2;
175 
176  return r;

189 void vf3_trits_from_bits(vf3_e *y, uint64_t *rnd) {
190  // assume rnd is loaded with 2 * len random words of 64 bits
191  int len = 1 + (y->alloc * sizeof(wave_word) - 1) / sizeof(uint64_t); // rounded up
192 
193  convert_word_t r0, r1, *y0, *y1;
194  y0 = rnd;
195  y1 = rnd + len;
196  for (int l = 0; l * sizeof(convert_word_t) < len * sizeof(uint64_t); ++l) {
197 #if defined(CONVERT_WORD_32)
198  r0 = convert32(y0[l]);
199  r1 = convert32(y1[l]);
200 #else // defined(CONVERT_WORD_64)
201  r0 = convert64(y0[l]);
202  r1 = convert64(y1[l]);
203 #endif
204  y0[l] = ((r0 >> 1) & EVEN) ^ (r1 & ODD);
205  y1[l] = (r0 & EVEN) ^ ((r1 << 1) & ODD) ^ y0[l];
206  }
207  memcpy(y->r0, y1, y->alloc * sizeof(wave_word));
208  memcpy(y->r1, y0, y->alloc * sizeof(wave_word));
