284 int decode(vf3_e *e, vf3_e *s, wave_sk_t *sk) {
285  int t, j;
286  uint16_t pi[N];
287 
288  // BEGIN alloc
289  vf3_e *y, *y_u, *y_v, *e_u, *e_v, *tmp;
290  y = vf3_alloc(N);
291  y_u = vf3_alloc(N2);
292  y_v = vf3_alloc(N2);
293  e_u = vf3_alloc(N2);
294  e_v = vf3_alloc(N2);
295  tmp = vf3_alloc(N);
296  // END alloc
297 
298  vf3_vector_cat_zero(tmp, s); // y = (s | 0)
299  memcpy(pi, sk->perm, N * sizeof(uint16_t)); // or whatever is suitable
300  vectperminv(y, tmp, pi); // now y is y permuted and pi is the inverse of sk->perm
301 
302  vf3_vector_split(y_u, y_v, y); // (y_u | y_v) <- y
303  vf3_vector_sub_mul_inplace(y_v, sk->c, y_u); // y_v = y_v - sk->c * y_u
304  vf3_vector_sub_mul_inplace(y_u, sk->b, y_v); // y_u = y_u - sk->b * y_v
305 
306  do {
307  t = decode_v(e_v, y_v, sk); // t the Hamming weight of e_v
308  j = decode_u(e_u, y_u, e_v, sk); // j the nb of (0,0) among matching pairs
309  } while (!accept_instance(t, j));
310  /*
311  rejection is deterministic, uncommon values of (t,j) are
312  rejected. The probability of occurrence is lower than 2^(-64),
313  but rejection is required for the security proof (tailcut lemma)
314  */
315 
316  vf3_vector_add_mul_inplace(e_u, sk->b, e_v); // e_u = e_u + sk->b * e_v
317  vf3_vector_add_mul_inplace(e_v, sk->c, e_u); // e_v = e_v + sk->c * e_u
318  vf3_vector_cat(tmp, e_u, e_v); // tmp = (e_u | e_v)
319  vectperminv(e, tmp, pi); // apply the inverse of pi, that is sk->perm, to tmp
320 
321  // BEGIN free
322  vf3_free(y);
323  vf3_free(y_u);
324  vf3_free(y_v);
325  vf3_free(e_u);
326  vf3_free(e_v);
327  vf3_free(tmp);
328  // END free
329  return 1;
