21  uint8_t a1 = (tx * tx) % 3; // a1 = 0 or 1
22  uint8_t a2 = 1 - a1; // a2 = 0 or 1
23  uint8_t a3 = (a2 - tx * ty + 6) % 3; // + 6, smallest multiple of 3 such that always a2 - tx * ty + 6 >= 0

44 void adhoc_reduce(vf3_e *x, vf3_e *y, int pos_l) {
45  uint8_t tx = vf3_get_element(pos_l, x);
46  uint8_t ty = vf3_get_element(pos_l, y);
47  uint8_t b = (tx * tx) % 3; // b = 0 or 1
48  uint8_t a = (1 - b - tx * ty + 6) % 3; // + 6, smallest multiple of 3 such that always 1 - b - tx * ty + 6 >= 0
49  wave_word maskb = -(b > 0);
50  wave_word maskcb = -(b == 0);
51  wave_word maska0 = -(a > 0);
52  wave_word maska1 = -(a > 1);
53  wave_word x0, x1, y0, y1, z0, z1, tmp;
54 
55  for (int i = 0; i < x->alloc; i++) {
56  // (x0, x1) <- i-th word of b * x
57  x0 = x->r0[i] & maskb;
58  x1 = x->r1[i] & x0;
59  // (y0, y1) <- i-th word of (1 - b) * y
60 // y0 = y->r0[i] & (maskb ^ 0xFFFFFFFFFFFFFFFFUL);
61  y0 = y->r0[i] & maskcb;
62  y1 = y->r1[i] & y0;
63  // (z0, z1) <- i-th word of b * x + (1 - b) * y
64  // (z0, z1) <- (x0, x1) + (y_0, y1)
65  tmp = x1 ^ y0;
66  z0 = (x0 ^ y0) | (tmp ^ y1);
67  z1 = (x0 ^ y1) & tmp;
68 
69  // (x0, x1) <- i-th word of a * x
70  x0 = x->r0[i] & maska0;
71  x1 = (x->r1[i] ^ maska1) & x0;
72  // (y0, y1) <- i-th word of b * y
73  y0 = y->r0[i] & maskb;
74  y1 = y->r1[i] & y0;
75  // i-th word of y <- i-th word of a * x + b * y
76  // i-th word of y <- (x0, x1) + (y0, y1)
77  tmp = x1 ^ y0;
78  y->r0[i] = (x0 ^ y0) | (tmp ^ y1);
79  y->r1[i] = (x0 ^ y1) & tmp;
80  // i-th word of x <- i-th word of b * x + (1 - b) * y
81  x->r0[i] = z0;
82  x->r1[i] = z1;
83  }
