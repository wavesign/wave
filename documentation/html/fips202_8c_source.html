20 /*************************************************
21  * Name: load64
22  *
23  * Description: Load 8 bytes into uint64_t in little-endian order
24  *
25  * Arguments: - const uint8_t *x: pointer to input byte array
26  *
27  * Returns the loaded 64-bit unsigned integer
28  **************************************************/
29 static uint64_t load64(const uint8_t *x) {
30  uint64_t r = 0;
31  for (size_t i = 0; i < 8; ++i) {
32  r |= (uint64_t) x[i] << 8 * i;
33  }
34 
35  return r;
36 }

340 /*************************************************
341  * Name: keccak_absorb
342  *
343  * Description: Absorb step of Keccak;
344  * non-incremental, starts by zeroeing the state.
345  *
346  * Arguments: - uint64_t *s: pointer to (uninitialized) output Keccak state
347  * - uint32_t r: rate in bytes (e.g., 168 for SHAKE128)
348  * - const uint8_t *m: pointer to input to be absorbed into s
349  * - size_t mlen: length of input in bytes
350  * - uint8_t p: domain-separation byte for different
351  * Keccak-derived functions
352  **************************************************/
353 static void keccak_absorb(uint64_t *s, uint32_t r, const uint8_t *m,
354  size_t mlen, uint8_t p) {
355  size_t i;
356  uint8_t t[200];
357 
358  /* Zero state */
359  for (i = 0; i < 25; ++i) {
360  s[i] = 0;
361  }
362 
363  while (mlen >= r) {
364  for (i = 0; i < r / 8; ++i) {
365  s[i] ^= load64(m + 8 * i);
366  }
367 
368  KeccakF1600_StatePermute(s);
369  mlen -= r;
370  m += r;
371  }
372 
373  for (i = 0; i < r; ++i) {
374  t[i] = 0;
375  }
376  for (i = 0; i < mlen; ++i) {
377  t[i] = m[i];
378  }
379  t[i] = p;
380  t[r - 1] |= 128;
381  for (i = 0; i < r / 8; ++i) {
382  s[i] ^= load64(t + 8 * i);
383  }
384 }

386 /*************************************************
387  * Name: keccak_squeezeblocks
388  *
389  * Description: Squeeze step of Keccak. Squeezes full blocks of r bytes each.
390  * Modifies the state. Can be called multiple times to keep
391  * squeezing, i.e., is incremental.
392  *
393  * Arguments: - uint8_t *h: pointer to output blocks
394  * - size_t nblocks: number of blocks to be
395  * squeezed (written to h)
396  * - uint64_t *s: pointer to input/output Keccak state
397  * - uint32_t r: rate in bytes (e.g., 168 for SHAKE128)
398  **************************************************/
399 static void keccak_squeezeblocks(uint8_t *h, size_t nblocks, uint64_t *s,
400  uint32_t r) {
401  while (nblocks > 0) {
402  KeccakF1600_StatePermute(s);
403  for (size_t i = 0; i < (r >> 3); i++) {
404  store64(h + 8 * i, s[i]);
405  }
406  h += r;
407  nblocks--;
408  }
409 }

430 /*************************************************
431  * Name: keccak_inc_absorb
432  *
433  * Description: Incremental keccak absorb
434  * Preceded by keccak_inc_init, succeeded by keccak_inc_finalize
435  *
436  * Arguments: - uint64_t *s_inc: pointer to input/output incremental state
437  * First 25 values represent Keccak state.
438  * 26th value represents either the number of absorbed bytes
439  * that have not been permuted, or not-yet-squeezed bytes.
440  * - uint32_t r: rate in bytes (e.g., 168 for SHAKE128)
441  * - const uint8_t *m: pointer to input to be absorbed into s
442  * - size_t mlen: length of input in bytes
443  **************************************************/
444 static void keccak_inc_absorb(uint64_t *s_inc, uint32_t r, const uint8_t *m,
445  size_t mlen) {
446  size_t i;
447 
448  /* Recall that s_inc[25] is the non-absorbed bytes xored into the state */
449  while (mlen + s_inc[25] >= r) {
450  for (i = 0; i < r - (uint32_t) s_inc[25]; i++) {
451  /* Take the i'th byte from message
452  xor with the s_inc[25] + i'th byte of the state; little-endian */
453  s_inc[(s_inc[25] + i) >> 3] ^= (uint64_t) m[i]
454  << (8 * ((s_inc[25] + i) & 0x07));
455  }
456  mlen -= (size_t) (r - s_inc[25]);
457  m += r - s_inc[25];
458  s_inc[25] = 0;
459 
460  KeccakF1600_StatePermute(s_inc);
461  }
462 
463  for (i = 0; i < mlen; i++) {
464  s_inc[(s_inc[25] + i) >> 3] ^= (uint64_t) m[i]
465  << (8 * ((s_inc[25] + i) & 0x07));
466  }
467  s_inc[25] += mlen;
468 }

470 /*************************************************
471  * Name: keccak_inc_finalize
472  *
473  * Description: Finalizes Keccak absorb phase, prepares for squeezing
474  *
475  * Arguments: - uint64_t *s_inc: pointer to input/output incremental state
476  * First 25 values represent Keccak state.
477  * 26th value represents either the number of absorbed bytes
478  * that have not been permuted, or not-yet-squeezed bytes.
479  * - uint32_t r: rate in bytes (e.g., 168 for SHAKE128)
480  * - uint8_t p: domain-separation byte for different
481  * Keccak-derived functions
482  **************************************************/
483 static void keccak_inc_finalize(uint64_t *s_inc, uint32_t r, uint8_t p) {
484  /* After keccak_inc_absorb, we are guaranteed that s_inc[25] < r,
485  so we can always use one more byte for p in the current state. */
486  s_inc[s_inc[25] >> 3] ^= (uint64_t) p << (8 * (s_inc[25] & 0x07));
487  s_inc[(r - 1) >> 3] ^= (uint64_t) 128 << (8 * ((r - 1) & 0x07));
488  s_inc[25] = 0;
489 }

491 /*************************************************
492  * Name: keccak_inc_squeeze
493  *
494  * Description: Incremental Keccak squeeze; can be called on byte-level
495  *
496  * Arguments: - uint8_t *h: pointer to output bytes
497  * - size_t outlen: number of bytes to be squeezed
498  * - uint64_t *s_inc: pointer to input/output incremental state
499  * First 25 values represent Keccak state.
500  * 26th value represents either the number of absorbed bytes
501  * that have not been permuted, or not-yet-squeezed bytes.
502  * - uint32_t r: rate in bytes (e.g., 168 for SHAKE128)
503  **************************************************/
504 static void keccak_inc_squeeze(uint8_t *h, size_t outlen, uint64_t *s_inc,
505  uint32_t r) {
506  size_t i;
507 
508  /* First consume any bytes we still have sitting around */
509  for (i = 0; i < outlen && i < s_inc[25]; i++) {
510  /* There are s_inc[25] bytes left, so r - s_inc[25] is the first
511  available byte. We consume from there, i.e., up to r. */
512  h[i] = (uint8_t) (s_inc[(r - s_inc[25] + i) >> 3]
513  >> (8 * ((r - s_inc[25] + i) & 0x07)));
514  }
515  h += i;
516  outlen -= i;
517  s_inc[25] -= i;
518 
519  /* Then squeeze the remaining necessary blocks */
520  while (outlen > 0) {
521  KeccakF1600_StatePermute(s_inc);
522 
523  for (i = 0; i < outlen && i < r; i++) {
524  h[i] = (uint8_t) (s_inc[i >> 3] >> (8 * (i & 0x07)));
525  }
526  h += i;
527  outlen -= i;
528  s_inc[25] = r - i;
529  }
530 }
